                <button on:click="pause()"><img src="time-out.png" width="25" height="25" alt="stop"></button>

<script>
    import {
        ipcRenderer,
        remote
    } from "electron";
    const {
        dialog,
        app
    } = remote
    const {
        readdir
    } = remote.require('fs-extra')

    import {
        resolve
    } from 'path'

    function handleName(name) {
        return name.replace('.mp3', '').split('-')[1]
    }

    export default {
        data() {
            return {
                files: [],
                title: ''
            };
        },
        helpers: {
            handleName
        },
        components: {
            Back: '../components/Back.svelte'
        },
        oncreate() {
            this.player = new Audio()
            document.querySelector('#app').appendChild(this.player)

            const {
                files
            } = this.store.get()

            this.set({
                files
            })

            this.player.onended = this.next.bind(this) // ç»“æŸè¿›å…¥ä¸‹ä¸€é¦–
            let listener = this.store.on('state', ({
                changed
            }) => {
                if (changed.index || changed.files) { // å½“é‡æ–°é€‰äº†æ–‡ä»¶ï¼Œæ’­æ”¾ç¬¬ä¸€é¦–
                    const {
                        files,
                        index
                    } = this.store.get()
                    this.player.src = "file://" + files[index]
                    this.set({
                        title: handleName(this.get().files[index])
                    })
                    this.play()
                }
            })
            this.on('destroy', listener.cancel)
        },
        ondestroy() {
            document.querySelectorAll("audio").forEach(a => a.remove())
        },
        methods: {
            next() {
                const {
                    index,
                    files
                } = this.store.get()
                let i = index + 1
                if (i > files.length - 1) {
                    i = files.length - 1
                }
                this.store.set({
                    index: i
                })
            },
            prev() {
                const {
                    index,
                    files
                } = this.store.get()
                let i = index - 1
                if (i < 0) {
                    i = 0
                }
                this.store.set({
                    index: i
                })
            },
            play() {
                if (!this.player.currentSrc) {
                    const {
                        files,
                        index
                    } = this.store.get()
                    this.set({
                        title: handleName(files[index])
                    })
                    this.player.src = "file://" + files[index]
                }
                this.player.play()
            },
            pause() {
                this.player.pause()
            },
            file() {
                dialog.showOpenDialog({
                    title: 'é€‰æ‹©éŸ³ä¹ç›®å½•',
                    defaultPath: app.getPath('home'),
                    properties: ['openDirectory']
                }, async(folders) => {
                    if (folders.length > 0) {
                        const folder = folders[0]
                        let files = await readdir(folder)
                        this.set({
                            files
                        })
                        const compare = (a, b) => {
                            let t1 = a.split('-')[0];
                            let t2 = b.split('-');
                            return parseInt(t1) - parseInt(t2)
                        }
                        files = files.sort(compare).map(file => resolve(folder, file))
                        this.store.set({
                            files,
                            index: 0
                        })
                    }
                })
            }
        }
    };
</script>
