function addDownloaed(plugins, local_plugins) {
  // åˆ¤æ–­ä¸‹è½½è¿‡çš„æ’ä»¶
  return plugins.map(plugin => {

      local_plugins.findIndex(

}

function inWorking(workingFilename, filename) {
